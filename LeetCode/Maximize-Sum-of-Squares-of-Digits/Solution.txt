class Solution {
    public String maxSumOfSquares(int n, int s) {
        if(s>9*n) return "";
        int d=s;
        StringBuilder r=new StringBuilder();
        for(int i=0;i<n;i++){
            int x=Math.min(9, d);
            r.append(x);
            d -=x;
        }
        if(d>0) return "";
        return r.toString();
    }
}
